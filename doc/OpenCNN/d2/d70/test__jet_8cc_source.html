112 template <typename Dtype>
113 Dtype jet_mul_div_add(const Dtype& x, const Dtype& y) {
114  Dtype z;
115  z = x * y / (x * x + y);
116  return z;
117  // from https://www.whitman.edu/mathematics/calculus_online/section14.03.html
118  /*
119  * dz/dx = (y*y - x*x*y)/((x*x+y)*(x*x+y))
120  * dz/dy = x*x*x/((x*x+y)*(x*x+y))
121  */
122 }
123 
124 template <typename Dtype>
125 Dtype jet_sub_div(const Dtype& x, const Dtype& y) {
126  Dtype z;
127  z = (x - y) / (x + y);
128  return z;
129  // http://math.gmu.edu/~memelian/teaching/Fall08/partDerivExamples.pdf
130  // 1 (e)
131  /*
132  * dz/dx = 2y/((x+y)*(x+y))
133  * dz/dy = -2x/((x+y)*(x+y))
134  */
135 }
