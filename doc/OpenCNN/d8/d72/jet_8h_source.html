101 template <typename Dtype, int N>
102 class Jet {
103  public:
108  Jet() : val_() { grad_.fill(0); }

121  Jet(const Dtype& val) // NOLINT
122  : val_(val) {
123  grad_.fill(0);

132  Jet(const Dtype& val, int i, Dtype derivative = 1) : val_(val) {
133  grad_.fill(0);
134  grad_.at(i) = derivative;
135  }
136 
143  void set(const Dtype& val, int i, Dtype derivative = 1) {
144  val_ = val;
145  grad_.fill(0);

147  }
148 
152  std::string to_string() const {
153  std::ostringstream ss;
154  ss << "[" << val_ << ", (";
155  std::string sep;
156  for (int i = 0; i < N; ++i) {
157  ss << sep << grad_[i];
158  sep = ", ";

160  ss << ")]";
161  ss << "\n";
162  return ss.str();
163  }
