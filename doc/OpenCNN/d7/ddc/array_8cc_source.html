86 template <typename Dtype>
87 void Array<Dtype>::init(int n, int c, int h, int w) {
88  CHECK_GE(n, 0);
89  CHECK_GE(c, 0);
90  CHECK_GE(h, 0);
91  CHECK_GE(w, 0);
92 
93  int total = n * c * h * w;
94  if (total == 0) {
95  if (d_) {
96  delete[] d_;
97  d_ = nullptr;
98  }
99  n_ = c_ = h_ = w_ = 0;
100  total_ = 0;
101  return;
102  }
103 
104  if (total != total_) {
105  if (d_) delete[] d_;
106 
107  d_ = new Dtype[n * c * h * w];
108  }
109 
110  memset(d_, 0, total * sizeof(Dtype));
111  n_ = n;
112  c_ = c;
113  h_ = h;
114  w_ = w;
115  total_ = total;
116 }
