104 void sgemv(const enum CBLAS_ORDER order, const enum CBLAS_TRANSPOSE TransA,
105  const int M, const int N, const float alpha, const float *A,
106  const int lda, const float *X, const int incX, const float beta,
107  float *Y, const int incY) {
108  if (TransA == CblasNoTrans) {
109  int num_rows = M;
110  int num_cols = N;
111  for (int r = 0; r < num_rows; ++r) {
112  float row_sum = 0;
113  for (int c = 0; c < num_cols; ++c) {
114  if (order == CblasRowMajor) {
115  // get the element of A at (r, c) for row major
116  float value = A[r * lda + c];
117  row_sum += value * X[c * incX];
118  } else {
119  // get the element of A at (r, c) for column major
120  float value = A[r + c * lda];
121  row_sum += value * X[c * incX];
122  }
123  }
124  *Y = alpha * row_sum + beta * (*Y);
125 
126  // process the next row
127  Y += incY;
128  }
129  } else {
130  // we omit the case for CblasConjTrans
131  // A is M-rows, N-cols
132  // trans(A) is N-rows, M-cols
133  // we want to compute y = alpha * trans(A) * x + beta * y
134 
135  int num_rows = N;
136  int num_cols = M;
137  for (int r = 0; r < num_rows; ++r) {
138  float row_sum = 0;
139  for (int c = 0; c < num_cols; ++c) {
140  float value;
141  if (order == CblasRowMajor) {
142  // get the element (r, c) of TransA, which is (c, r) of A
143  value = A[c * lda + r];
144  } else {
145  // for column major
146  // get the element (r, c) of TransA, which is (c, r) of A
147  value = A[c + r * lda];
148  }
149  row_sum += value * X[c * incX];
150  }
151  *Y = alpha * row_sum + beta * (*Y);
152 
153  // process the next row
154  Y += incY;
155  }
156  }
